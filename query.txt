#channel_values in json
SELECT
  checkpoint_id,
  checkpoint->'channel_values' AS channel_values
FROM checkpoints
WHERE thread_id = 'reg_demo_1'
ORDER BY checkpoint_id DESC
LIMIT 10;

#blobs
SELECT
  channel,
  version,
  type,
  length(blob) AS blob_len,
  left(encode(blob, 'escape'), 160) AS blob_preview
FROM checkpoint_blobs
WHERE thread_id='reg_demo_1'
ORDER BY version DESC
LIMIT 25;
